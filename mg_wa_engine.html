How to get your API credentials (step by step)

1Create Meta Business Account

Go to business.facebook.com → Create a Business Account. This is free. You'll need a Facebook account.

2Add WhatsApp to your Business

In Meta Business Manager → Settings → Accounts → WhatsApp Accounts → Add. Link your WhatsApp Business number.

3Create a Meta App

Go to developers.facebook.com → Create App → Business type. Add "WhatsApp" product to your app.

4Get your Phone Number ID

In your Meta App → WhatsApp → Getting Started → copy Phone Number ID and WhatsApp Business Account ID.

5Create permanent access token

Meta Business Manager → Settings → System Users → Add System User (Admin) → Generate Token → Select your App → Give whatsapp_business_messaging permission. Copy the token.

6Setup webhook for auto-replies

In your Meta App → WhatsApp → Configuration → Webhook URL: https://YOUR-SERVER/webhook
Verify Token: whatever you set above. Subscribe to: messages.
Note: You need a public HTTPS server for live webhook. For testing, use ngrok.io

✓Free tier limits

Meta gives you 1,000 free conversations/month (business-initiated). After that it's ~₹0.50–2 per conversation. One conversation = 24 hours of messages to one person.

Webhook server code (Node.js — paste & deploy)

// Save as server.js, run: node server.js
// Deploy free on: Railway.app, Render.com, or Vercel

const express = require('express');
const app = express();
app.use(express.json());

const VERIFY_TOKEN = 'moneygroww_secret_2025'; // match your verify token
const WA_TOKEN = process.env.WA_TOKEN; // set in environment
const PHONE_ID = process.env.PHONE_ID; // set in environment

// Webhook verification
app.get('/webhook', (req, res) => {
  if (req.query['hub.verify_token'] === VERIFY_TOKEN) {
    res.send(req.query['hub.challenge']);
  } else res.sendStatus(403);
});

// Receive messages & status updates
app.post('/webhook', async (req, res) => {
  res.sendStatus(200); // respond fast
  const data = req.body;
  const entry = data.entry?.[0]?.changes?.[0]?.value;
  if (!entry) return;

  // INCOMING MESSAGE from customer
  if (entry.messages) {
    const msg = entry.messages[0];
    const from = msg.from; // customer's phone
    const text = msg.text?.body || '';
    console.log('INBOUND from', from, ':', text);
    
    // Auto-reply with AI chatbot
    const reply = getBotReply(text, from);
    await sendWA(from, reply);
  }

  // DELIVERY STATUS updates
  if (entry.statuses) {
    const st = entry.statuses[0];
    // st.status = 'sent', 'delivered', 'read', 'failed'
    // st.id = message_id, st.recipient_id = customer phone
    console.log('STATUS:', st.status, 'for', st.recipient_id);
    // Update your database here
  }
});

// Send a WhatsApp message
async function sendWA(to, text) {
  const res = await fetch(
    `https://graph.facebook.com/v18.0/${PHONE_ID}/messages`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${WA_TOKEN}`
      },
      body: JSON.stringify({
        messaging_product: 'whatsapp',
        to: to,
        type: 'text',
        text: { body: text }
      })
    }
  );
  return res.json();
}

// Send template message (for first outreach, required by Meta)
async function sendTemplate(to, templateName, langCode, components) {
  const res = await fetch(
    `https://graph.facebook.com/v18.0/${PHONE_ID}/messages`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${WA_TOKEN}`
      },
      body: JSON.stringify({
        messaging_product: 'whatsapp',
        to: to,
        type: 'template',
        template: {
          name: templateName,
          language: { code: langCode || 'en' },
          components: components || []
        }
      })
    }
  );
  return res.json();
}

function getBotReply(msg, from) {
  const m = msg.toLowerCase();
  if (m.includes('home') || m.includes('house'))
    return '🏠 *Home Loan from MoneyGroww:*\n\n✅ Rate from 8.40% p.a.\n✅ Up to ₹5 Crore\n✅ 3-day approval\n\nReply *YES* for free consultation! 📞';
  if (m.includes('business'))
    return '💼 *Business Loan:*\n\n✅ Rate from 11% p.a.\n✅ Up to ₹5 Crore\n✅ Quick disbursal\n\nReply *YES* for callback! 🚀';
  if (m.includes('yes') || m.includes('call') || m.includes('interested'))
    return '✅ *You are on our priority list!*\n\nOur expert will call you within 2 hours.\n📞 Direct: +91 98765 43210\n\n_MoneyGroww — Your loan, your way!_ 💚';
  return '👋 Hello! I\'m your MoneyGroww AI assistant.\n\nAsk about:\n🏠 Home Loan\n💼 Business Loan\n👤 Personal Loan\n🏢 LAP\n\nOr reply *CALL* for expert callback! 📞';
}

app.listen(3000, () => console.log('MoneyGroww WA server running on port 3000'));
